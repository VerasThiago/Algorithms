Segment Tree

int seg[1000000];
int v[10000000];
void SegInit(int idx, int L, int R){
  if(L == R) seg[idx] = v[L];
  else{
    int mid = (L+R)/2;
    SegInit(idx*2,L,mid);
    SegInit((idx*2) + 1,mid+1, R);
    seg[idx] = seg[idx*2] + seg[(idx*2) + 1];
  }
}

int find(int idx, int L, int R, int S, int E){
  if(S <= L and R <= E) return seg[idx];
  if(L > E or S > R) return 0;
  return (find(idx*2, L, (L+R)/2, S, E) + find(idx*2 + 1, (L+R)/2 + 1, R, S, E));

}
void update(int idx, int L , int R,int valor, int quem){
  if( !(L <= quem and quem <= R) ) return;  
  seg[idx] -= valor;
  if(L == R) return;
  int mid = (L+R)/2;
  update(idx*2, L, mid, valor, quem);
  update(idx*2 + 1, mid+1, R, valor,quem);
}

bool PesquisaBinaria ( int k[], int who , int n){
  int inf,sup,meio;
  inf=0;
  sup=n-1;
  while (inf<=sup)
  {
    meio=(inf+sup)/2;
    if (who==k[meio])
         return true;
    else if (who< k[meio])
         sup=meio-1;
    else
         inf=meio+1;
  }
  return false;  
}

Coin Change Algorithm

int coin(int troco, vector moedas){
  vector ans(troco+1,1e9+7);
  ans[0] = 0;
  for(int i = 1 ; i <= troco; i++){
    for(int j = 0 ; j < moedas.size(); j++){
      if(i < moedas[j]) break;
      ans[i] = min(ans[i],1+ans[i - moedas[j]]);
    }
  }
  return ans[troco];
}

Edit Distance

double st[MAX_M][MAX_N];
double c_i = 2.5, c_r = 0.5, c_s = 1;

double edit(const string& s, const string& r)
{
    int m = s.size();
    int n = r.size();

    for (int i = 0; i <= m; ++i)
        st[i][0] = i*c_r;

    for (int j = 1; j <= n; ++j)
        st[0][j] = j*c_i;

    for (int i = 1; i <= m; ++i)
        for (int j = 1; j <= n; ++j)
        {
            double insertion = st[i][j - 1] + c_i;
            double deletion = st[i - 1][j] + c_r;
            double change = st[i - 1][j - 1] + c_s * (s[i - 1] == r[j - 1] ? 0 : 1);
            st[i][j] = min(insertion, deletion);
            st[i][j] = min(st[i][j], change);
        }

    return st[m][n];
}

Matriz mul(int n , Matriz* A, Matriz* B){

  Matriz C;
    for(i = 0 ; i < n ; i++)
      for(j = 0 ; j < n ; j++){
        C.m[i][j] = 0;
        for(k = 0 ; k < n ; k++)
          C.m[i][j] = (C.m[i][j] + (A->m[i][k]%13 * B->m[k][j]%13)%13)%13;

Matriz power(int n, Matriz base, long long int expo){
  int i , j;
  Matriz resp;

  for(i = 0 ; i < n ; i++)
    for(j = 0 ; j < n ; j++){
      if( i == j)
        resp.m[i][j] = 1;
      else
         resp.m[i][j] = 0;
    }

  while(expo > 0){
    if( expo%2 == 1)
      resp = mul(n, &resp, &base);
    base = mul(n, &base, &base);
    expo /= 2;
  }
  return resp;  
}

Z Function

vector z_function(const string &s) {
    int n = s.size();
    vector z(n, 0);
    int L = 0, R = 0;
    for(int i = 1; i < n; i++) {
        if(i <= R) {
            z[i] = min(z[i - L], R - i + 1);
        }
        while(z[i] + i < n && s[z[i] + i] == s[z[i]]) {
            z[i]++;
        }
        if(R < i + z[i] - 1) {
            L = i, R = i + z[i] - 1;
        }
    }
    return z;
}
